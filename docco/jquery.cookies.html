$.cookies.get = function(name, /* optional */ subCookie)
{
    var cookies = new Cookies();
    var cookie = cookies[name];
    if (cookie)
    {
        if (subCookie)
        {
            if (cookie.subCookies)
            {
                return cookie.subCookies[subCookie] || null;
            }

            return null;
        }

        if (cookie.subCookies)
        {
            return cookie.subCookies;
        }
        else
        {
            return cookie.value || "";
        }
    }
    
    return null;
